@echo off
setlocal EnableExtensions

set "SCRIPT_DIR=%~dp0"
set "LOCAL_PS1=%SCRIPT_DIR%setup_proxy.ps1"
set "REMOTE_URL=https://raw.githubusercontent.com/chenzai666/proxy-setup/master/setup_proxy.ps1"

if defined PROXY_SETUP_REMOTE_URL set "REMOTE_URL=%PROXY_SETUP_REMOTE_URL%"

echo.
echo ==========
echo   Windows CMD Proxy Setup
echo ==========
echo.

if exist "%LOCAL_PS1%" (
    echo   [OK] Using local setup_proxy.ps1
    set "PROXY_SETUP_LOCAL_PS1=%LOCAL_PS1%"
    powershell.exe -NoProfile -ExecutionPolicy Bypass -Command "$p=$env:PROXY_SETUP_LOCAL_PS1; $code=[IO.File]::ReadAllText($p,[Text.Encoding]::UTF8); iex $code"
    exit /b %ERRORLEVEL%
)

echo   Local setup_proxy.ps1 not found.
echo   Downloading: %REMOTE_URL%
echo.

set "PROXY_SETUP_PS1_URL=%REMOTE_URL%"
powershell.exe -NoProfile -ExecutionPolicy Bypass -Command "[Net.ServicePointManager]::SecurityProtocol=[Net.SecurityProtocolType]::Tls12; $w=New-Object Net.WebClient; $w.Encoding=[Text.Encoding]::UTF8; iex($w.DownloadString($env:PROXY_SETUP_PS1_URL))"
exit /b %ERRORLEVEL%
